# Schema-driven extraction prompt for contract doc-type

doc_type: contract
parser_hint: "layout"

vocabulary_refs:
  - global.legal
  - global.commercial_india

fields:
  jurisdiction:
    description: "Governing legal jurisdiction"
    type: string | enum
    locator_hint: "explicit jurisdiction section OR implicit 'governed by laws of X' pattern"     ← edited
    fallback_synonyms: ["laws of", "governed by", "subject to the jurisdiction of"]     ← added
    on_low_conf: review

  party_a:
    description: "First named party (procurer / employer / lessor)"
    type: entity_ref(org)
    locator_hint: "first paragraph after 'between'"
    required: true
